159 def cvxopt_foopsi(fluor, b, c1, g, sn, p, bas_nonneg, verbosity):
160  """Solve the deconvolution problem using cvxopt and picos packages
161  """
162  try:
163  from cvxopt import matrix, spmatrix, spdiag, solvers
164  import picos
165  except ImportError:
166  raise ImportError('Constrained Foopsi requires cvxopt and picos packages.')
167 
168  T = len(fluor)
169 
170  # construct deconvolution matrix (sp = G*c)
171  G = spmatrix(1., list(range(T)), list(range(T)), (T, T))
172 
173  for i in range(p):
174  G = G + spmatrix(-g[i], np.arange(i + 1, T), np.arange(T - i - 1), (T, T))
175 
176  gr = np.roots(np.concatenate([np.array([1]), -g.flatten()]))
177  gd_vec = np.max(gr)**np.arange(T) # decay vector for initial fluorescence
178  gen_vec = G * matrix(np.ones(fluor.size))
179 
180  # Initialize variables in our problem
181  prob = picos.Problem()
182 
183  # Define variables
184  calcium_fit = prob.add_variable('calcium_fit', fluor.size)
185  cnt = 0
186  if b is None:
187  flag_b = True
188  cnt += 1
189  b = prob.add_variable('b', 1)
190  if bas_nonneg:
191  b_lb = 0
192  else:
193  b_lb = np.min(fluor)
194 
195  prob.add_constraint(b >= b_lb)
196  else:
197  flag_b = False
198 
199  if c1 is None:
200  flag_c1 = True
201  cnt += 1
202  c1 = prob.add_variable('c1', 1)
203  prob.add_constraint(c1 >= 0)
204  else:
205  flag_c1 = False
206 
207  # Add constraints
208  prob.add_constraint(G * calcium_fit >= 0)
209  res = abs(matrix(fluor.astype(float)) - calcium_fit - b *
210  matrix(np.ones(fluor.size)) - matrix(gd_vec) * c1)
211  prob.add_constraint(res < sn * np.sqrt(fluor.size))
212  prob.set_objective('min', calcium_fit.T * gen_vec)
213 
214  # solve problem
215  try:
216  prob.solve(solver='mosek', verbose=verbosity)
217  sel_solver = 'mosek'
218 # prob.solve(solver='gurobi', verbose=verbosity)
219 # sel_solver = 'gurobi'
220  except ImportError:
221  warn('MOSEK is not installed. Spike inference may be VERY slow!')
222  sel_solver = []
223  prob.solver_selection()
224  prob.solve(verbose=verbosity)
225 
226  # if problem in infeasible due to low noise value then project onto the
227  # cone of linear constraints with cvxopt
228  if prob.status == 'prim_infeas_cer' or prob.status == 'dual_infeas_cer' or prob.status == 'primal infeasible':
229  warn('Original problem infeasible. Adjusting noise level and re-solving')
230  # setup quadratic problem with cvxopt
231  solvers.options['show_progress'] = verbosity
232  ind_rows = list(range(T))
233  ind_cols = list(range(T))
234  vals = np.ones(T)
235  if flag_b:
236  ind_rows = ind_rows + list(range(T))
237  ind_cols = ind_cols + [T] * T
238  vals = np.concatenate((vals, np.ones(T)))
239  if flag_c1:
240  ind_rows = ind_rows + list(range(T))
241  ind_cols = ind_cols + [T + cnt - 1] * T
242  vals = np.concatenate((vals, np.squeeze(gd_vec)))
243  P = spmatrix(vals, ind_rows, ind_cols, (T, T + cnt))
244  H = P.T * P
245  Py = P.T * matrix(fluor.astype(float))
246  sol = solvers.qp(
247  H, -Py, spdiag([-G, -spmatrix(1., list(range(cnt)), list(range(cnt)))]), matrix(0., (T + cnt, 1)))
248  xx = sol['x']
249  c = np.array(xx[:T])
250  sp = np.array(G * matrix(c))
251  c = np.squeeze(c)
252  if flag_b:
253  b = np.array(xx[T + 1]) + b_lb
254  if flag_c1:
255  c1 = np.array(xx[-1])
256  sn = old_div(np.linalg.norm(fluor - c - c1 * gd_vec - b), np.sqrt(T))
257  else: # readout picos solution
258  c = np.squeeze(calcium_fit.value)
259  sp = np.squeeze(np.asarray(G * calcium_fit.value))
260  if flag_b:
261  b = np.squeeze(b.value)
262  if flag_c1:
263  c1 = np.squeeze(c1.value)
264 
265  return c, b, c1, g, sn, sp

399 def _nnls(KK, Ky, s=None, mask=None, tol=1e-9, max_iter=None):
400  """
401  Solve non-negative least squares problem
402  ``argmin_s || Ks - y ||_2`` for ``s>=0``
403 
404  Parameters
405  ----------
406  KK : array, shape (n, n)
407  Dot-product of design matrix K transposed and K, K'K
408  Ky : array, shape (n,)
409  Dot-product of design matrix K transposed and target vector y, K'y
410  s : None or array, shape (n,), optional, default None
411  Initialization of deconvolved neural activity.
412  mask : array of bool, shape (n,), optional, default (True,)*n
413  Mask to restrict potential spike times considered.
414  tol : float, optional, default 1e-9
415  Tolerance parameter.
416  max_iter : None or int, optional, default None
417  Maximum number of iterations before termination.
418  If None (default), it is set to len(KK).
419 
420  Returns
421  -------
422  s : array, shape (n,)
423  Discretized deconvolved neural activity (spikes)
424 
425  References
426  ----------
427  * Lawson C and Hanson RJ, SIAM 1987
428  * Bro R and DeJong S, J Chemometrics 1997
429  """
430 
431  if mask is None:
432  mask = np.ones(len(KK), dtype=bool)
433  else:
434  KK = KK[mask][:, mask]
435  Ky = Ky[mask]
436  if s is None:
437  s = np.zeros(len(KK))
438  l = Ky.copy()
439  P = np.zeros(len(KK), dtype=bool)
440  else:
441  s = s[mask]
442  P = s > 0
443  l = Ky - KK[:, P].dot(s[P])
444  i = 0
445  if max_iter is None:
446  max_iter = len(KK)
447  for i in range(max_iter): # max(l) is checked at the end, should do at least one iteration
448  w = np.argmax(l)
449  P[w] = True
450  try: # likely unnnecessary try-except-clause for robustness sake
451  mu = np.linalg.inv(KK[P][:, P]).dot(Ky[P])
452  except:
453  mu = np.linalg.inv(KK[P][:, P] + tol * np.eye(P.sum())).dot(Ky[P])
454  print(r'added $\epsilon$I to avoid singularity')
455  while len(mu > 0) and min(mu) < 0:
456  a = min(s[P][mu < 0] / (s[P][mu < 0] - mu[mu < 0]))
457  s[P] += a * (mu - s[P])
458  P[s <= tol] = False
459  try:
460  mu = np.linalg.inv(KK[P][:, P]).dot(Ky[P])
461  except:
462  mu = np.linalg.inv(KK[P][:, P] + tol * np.eye(P.sum())).dot(Ky[P])
463  print(r'added $\epsilon$I to avoid singularity')
464  s[P] = mu.copy()
465  l = Ky - KK[:, P].dot(s[P])
466  if max(l) < tol:
467  break
468  tmp = np.zeros(len(mask))
469  tmp[mask] = s
470  return tmp

473 def onnls(y, g, lam=0, shift=100, window=None, mask=None, tol=1e-9, max_iter=None):
474  """ Infer the most likely discretized spike train underlying an AR(2) fluorescence trace
475 
476  Solves the sparse non-negative deconvolution problem
477  ``argmin_s 1/2|Ks-y|^2 + lam |s|_1`` for ``s>=0``
478 
479  Parameters
480  ----------
481  y : array of float, shape (T,)
482  One dimensional array containing the fluorescence intensities with
483  one entry per time-bin.
484  g : array, shape (p,)
485  if p in (1,2):
486  Parameter(s) of the AR(p) process that models the fluorescence impulse response.
487  else:
488  Kernel that models the fluorescence impulse response.
489  lam : float, optional, default 0
490  Sparsity penalty parameter lambda.
491  shift : int, optional, default 100
492  Number of frames by which to shift window from on run of NNLS to the next.
493  window : int, optional, default None (200 or larger dependend on g)
494  Window size.
495  mask : array of bool, shape (n,), optional, default (True,)*n
496  Mask to restrict potential spike times considered.
497  tol : float, optional, default 1e-9
498  Tolerance parameter.
499  max_iter : None or int, optional, default None
500  Maximum number of iterations before termination.
501  If None (default), it is set to window size.
502 
503  Returns
504  -------
505  c : array of float, shape (T,)
506  The inferred denoised fluorescence signal at each time-bin.
507  s : array of float, shape (T,)
508  Discretized deconvolved neural activity (spikes).
509 
510  References
511  ----------
512  * Friedrich J and Paninski L, NIPS 2016
513  * Bro R and DeJong S, J Chemometrics 1997
514  """
515 
516  T = len(y)
517  if mask is None:
518  mask = np.ones(T, dtype=bool)
519  if window is None:
520  w = max(200, len(g) if len(g) > 2 else
521  int(-5 / log(g[0] if len(g) == 1 else
522  (g[0] + sqrt(g[0] * g[0] + 4 * g[1])) / 2)))
523  else:
524  w = window
525  w = min(T, w)
526  K = np.zeros((w, w))
527  if len(g) == 1: # kernel for AR(1)
528  _y = y - lam * (1 - g[0])
529  _y[-1] = y[-1] - lam
530  h = np.exp(log(g[0]) * np.arange(w))
531  for i in range(w):
532  K[i:, i] = h[:w - i]
533  elif len(g) == 2: # kernel for AR(2)
534  _y = y - lam * (1 - g[0] - g[1])
535  _y[-2] = y[-2] - lam * (1 - g[0])
536  _y[-1] = y[-1] - lam
537  d = (g[0] + sqrt(g[0] * g[0] + 4 * g[1])) / 2
538  r = (g[0] - sqrt(g[0] * g[0] + 4 * g[1])) / 2
539  if d == r:
540  h = np.exp(log(d) * np.arange(1, w + 1)) * np.arange(1, w + 1)
541  else:
542  h = (np.exp(log(d) * np.arange(1, w + 1)) -
543  np.exp(log(r) * np.arange(1, w + 1))) / (d - r)
544  for i in range(w):
545  K[i:, i] = h[:w - i]
546  else: # arbitrary kernel
547  h = g
548  for i in range(w):
549  K[i:, i] = h[:w - i]
550  a = np.linalg.inv(K).sum(0)
551  _y = y - lam * a[0]
552  _y[-w:] = y[-w:] - lam * a
553 
554  s = np.zeros(T)
555  KK = K.T.dot(K)
556  for i in range(0, max(1, T - w), shift):
557  s[i:i + w] = _nnls(KK, K.T.dot(_y[i:i + w]), s[i:i + w], mask=mask[i:i + w],
558  tol=tol, max_iter=max_iter)[:w]
559  # subtract contribution of spikes already committed to
560  _y[i:i + w] -= K[:, :shift].dot(s[i:i + shift])
561  s[i + shift:] = _nnls(KK[-(T - i - shift):, -(T - i - shift):],
562  K[:T - i - shift, :T - i - shift].T.dot(_y[i + shift:]),
563  s[i + shift:], mask=mask[i + shift:])
564  c = np.zeros_like(s)
565  for t in np.where(s > tol)[0]:
566  c[t:t + w] += s[t] * h[:min(w, T - t)]
567  return c, s

570 def constrained_oasisAR2(y, g, sn, optimize_b=True, b_nonneg=True, optimize_g=0, decimate=5,
571  shift=100, window=None, tol=1e-9, max_iter=1, penalty=1):
572  """ Infer the most likely discretized spike train underlying an AR(2) fluorescence trace
573 
574  Solves the noise constrained sparse non-negative deconvolution problem
575  min |s|_1 subject to |c-y|^2 = sn^2 T and s_t = c_t-g1 c_{t-1}-g2 c_{t-2} >= 0
576 
577  Parameters
578  ----------
579  y : array of float
580  One dimensional array containing the fluorescence intensities (with baseline
