32 def com(A, d1, d2):
33  """Calculation of the center of mass for spatial components
34  Inputs:
35  A: np.ndarray
36  matrix of spatial components (d x K)
37  d1: int
38  number of pixels in x-direction
39  d2: int
40  number of pixels in y-direction
41 
42  Output:
43  cm: np.ndarray
44  center of mass for spatial components (K x 2)
45  """
46  nr = np.shape(A)[-1]
47  Coor = dict()
48  Coor['x'] = np.kron(np.ones((d2, 1)), np.expand_dims(list(range(d1)), axis=1))
49  Coor['y'] = np.kron(np.expand_dims(list(range(d2)), axis=1), np.ones((d1, 1)))
50  cm = np.zeros((nr, 2)) # vector for center of mass
51  cm[:, 0] = old_div(np.dot(Coor['x'].T, A), A.sum(axis=0))
52  cm[:, 1] = old_div(np.dot(Coor['y'].T, A), A.sum(axis=0))
53 
54  return cm

196 def distance_masks(M_s,cm_s,max_dist):
197  """
198  Compute distance matrix based on an intersection over union metric. Matrix are compared in order, with matrix i compared with matrix i+1
199 
200  Parameters
201  ----------
202  M_s: list of ndarrays
203  The thresholded A matrices (masks) to compare, output of threshold_components
204 
205  cm_s: list of list of 2-ples
206  the centroids of the components in each M_s
207 
208  max_dist: float
209  maximum distance among centroids allowed between components. This corresponds to a distance at which two components are surely disjoined
210 
211 
212 
213  Returns:
214  --------
215  D_s: list of matrix distances
216 
217  """
218  D_s=[]
219 
220  for M1,M2,cm1,cm2 in zip(M_s[:-1],M_s[1:],cm_s[:-1],cm_s[1:]):
221  #better with a function that calls itself
222  print('New Pair **')
223  #not to interfer with M_s
224  M1= M1.copy()[:,:]
225  M2= M2.copy()[:,:]
226  #take the last dimension's number (x*y)
227  d_1=np.shape(M1)[-1]
228  d_2=np.shape(M2)[-1]
229  D = np.ones((d_1,d_2));
230 
231  cm1=np.array(cm1)
232  cm2=np.array(cm2)
233  for i in range(d_1): #for each components of gt
234  k=M1[:,np.repeat(i,d_2)]+M2 # k is correlation matrix of this neuron to every other of the test
235  # h=M1[:,np.repeat(i,d_2)].copy()
236  # h.multiply(M2)
237  for j in range(d_2): #for each components on the tests
238  #
239  dist = np.linalg.norm(cm1[i]-cm2[j]) # we compute the distance of this one to the other ones
240  if dist<max_dist:
241  #union matrix of the ith neuron to the jth one
242  union = k[:,j].sum()
243  #we could have used OR for union and AND for intersection while converting the matrice into real boolean before
244  # intersection = h[:,j].nnz
245  #product of the two elements' matrices
246  #we multiply the boolean values from the jth omponent to the ith
247  intersection= np.array(M1[:,i].T.dot(M2[:,j]).todense()).squeeze()
248 
251  if union > 0:
252  #intersection is removed from union since union contains twice the overlaping area
253  #having the values in this format 0-1 is helpfull for the hungarian algorithm that follows
254  D[i,j] = 1-1.*intersection/(union-intersection)
255  else:
256 # print 'empty component: setting distance to max'
257  D[i,j] = 1.
258 
259  if np.isnan(D[i,j]):
260  raise Exception('Nan value produced. Error in inputs')
261  else:
262  D[i,j] = 1
263 
264  D_s.append(D)
265  return D_s

268 def find_matches(D_s, print_assignment=False):
269 
270  matches=[]
271  costs=[]
272  t_start=time.time()
273  for ii,D in enumerate(D_s):
274  #we make a copy not to set changes in the original
275  DD=D.copy()
276  if np.sum(np.where(np.isnan(DD)))>0:
277  raise Exception('Distance Matrix contains NaN, not allowed!')
278 
279 
280  # indexes = m.compute(DD)
281 # indexes = linear_assignment(DD)
282  #we do the hungarian
283  indexes = linear_sum_assignment(DD)
284  indexes2=[(ind1,ind2) for ind1,ind2 in zip(indexes[0],indexes[1])]
285  matches.append(indexes)
286  DD=D.copy()
287  total = []
288  #we want to extract those informations from the hungarian algo
289  for row, column in indexes2:
290  value = DD[row,column]
291  if print_assignment:
292  print(('(%d, %d) -> %f' % (row, column, value)))
293  total.append(value)
294  print(('FOV: %d, shape: %d,%d total cost: %f' % (ii, DD.shape[0],DD.shape[1], np.sum(total))))
295  print((time.time()-t_start))
296  costs.append(total)
297  #we send back the results in the format we want
298  return matches,costs
