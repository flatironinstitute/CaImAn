318 def extract_DF_F(Yr, A, C, bl, quantileMin = 8, frames_window = 200, block_size = 400, dview = None):
319  """ Compute DFF function from cnmf output.
320 
321  Disclaimer: it might be memory inefficient
322 
323  Parameters:
324  -----------
325  Yr: ndarray (2D)
326  movie pixels X time
327 
328  A: scipy.sparse.coo_matrix
329  spatial components (from cnmf cnm.A)
330 
331  C: ndarray
332  temporal components (from cnmf cnm.C)
333 
334  bl: ndarray
335  baseline for each component (from cnmf cnm.bl)
336 
337  quantile_min: float
338  quantile minimum of the
339 
340  frammes_window: int
341  number of frames for running quantile
342 
343  Returns:
344  -------
345 
346  Cdf:
347  the computed Calcium acitivty to the derivative of f
348 
349  See Also:
350  -------
351 
352  ..image::docs/img/onlycnmf.png
353 
354  """
355  nA = np.array(np.sqrt(A.power(2).sum(0)).T)
356  A = scipy.sparse.coo_matrix(A/nA.T)
357  C = C*nA
358  bl = (bl*nA.T).squeeze()
359  nA = np.array(np.sqrt(A.power(2).sum(0)).T)
360 
361 
362  T = C.shape[-1]
363  if 'memmap' in str(type(Yr)):
364  if block_size >= 500:
365  print('Forcing single thread for memory issues')
366  dview_res = None
367  else:
368  print('Using thread. If memory issues set block_size larger than 500')
369  dview_res = dview
370 
371  AY = parallel_dot_product(Yr, A, dview=dview_res, block_size=block_size,
372  transpose=True).T
373  else:
374  AY = A.T.dot(Yr)
375 
376  bas_val = bl[None,:]
377  Bas = np.repeat(bas_val,T,0).T
378  AA = A.T.dot(A)
379  AA.setdiag(0)
380  Cf = (C-Bas)*(nA**2)
381  C2 = AY - AA.dot(C)
382 
383 
384  if frames_window is None or frames_window > T:
385  Df = np.percentile(C2,quantileMin, axis = 1)
386  C_df = Cf/Df[:,None]
387 
388  else:
389  Df = scipy.ndimage.percentile_filter(C2, quantileMin, (frames_window,1))
390  C_df = Cf/Df
391 
392  return C_df

527 def app_vertex_cover(A):
528  """ Finds an approximate vertex cover for a symmetric graph with adjacency matrix A.
529 
530  Parameters:
531  -----------
532  A: boolean 2d array (K x K)
533  Adjacency matrix. A is boolean with diagonal set to 0
534 
535  Returns:
536  --------
537  L: A vertex cover of A
538 
539  @authors by Eftychios A. Pnevmatikakis, Simons Foundation, 2015
540  """
541 
542  L = []
543  while A.any():
544  nz = np.nonzero(A)[0] # find non-zero edges
545  u = nz[np.random.randint(0, len(nz))]
546  A[u, :] = False
547  A[:, u] = False
548  L.append(u)
549 
550  return np.asarray(L)

553 def update_order(A):
554  """Determines the update order of the temporal components given the spatial
555 
556  components by creating a nest of random approximate vertex covers
557 
558  Input:
559  -------
560  A: np.ndarray
561  matrix of spatial components (d x K)
562 
563  Outputs:
564  ---------
565  O: list of sets
566  list of subsets of components. The components of each subset can be updated in parallel
567 
568  lo: list
569  length of each subset
570 
571  @authors: Eftychios A. Pnevmatikakis, Simons Foundation, 2015
572  """
573  K = np.shape(A)[-1]
574  AA = A.T * A
575  AA.setdiag(0)
576  F = (AA) > 0
577  F = F.toarray()
578  rem_ind = np.arange(K)
579  O = []
580  lo = []
581  while len(rem_ind) > 0:
582  L = np.sort(app_vertex_cover(F[rem_ind, :][:, rem_ind]))
583  if L.size:
584  ord_ind = set(rem_ind) - set(rem_ind[L])
585  rem_ind = rem_ind[L]
586  else:
587  ord_ind = set(rem_ind)
588  rem_ind = []
589 
590  O.append(ord_ind)
591  lo.append(len(ord_ind))
592 
593  return O[::-1], lo[::-1]

596 def update_order_greedy(A,flag_AA = True):
597  """Determines the update order of the temporal components given the spatial
598 
599  components using a greedy method
600 
601  Input:
602  -------
603  A: sparse crc matrix
604  matrix of spatial components (d x K)
605  OR
606  A.T.dot(A) matrix (d x d) if flag_AA = true
607 
608  flag_AA: boolean (default true)
609 
610  Outputs:
611  ---------
612  O: list of sets
613  list of subsets of components. The components of each subset can be updated in parallel
614 
615  lo: list
616  length of each subset
617 
618  @author: Eftychios A. Pnevmatikakis, Simons Foundation, 2017
619  """
620  K = np.shape(A)[-1]
621  O = []
622  for i in range(K):
623  new_list = True
624  for ls in O:
625  if flag_AA:
626  if A[i,ls].nnz == 0:
627  ls.append(i)
628  new_list = False
629  break
630  else:
631  if (A[:,i].T.dot(A[:,ls])).nnz == 0:
632  ls.append(i)
633  new_list = False
634  break
635 
636  if new_list:
637  O.append([i])
638 
639  lo = [len(ls) for ls in O]
640  return O,lo

645 def order_components(A, C):
646  """Order components based on their maximum temporal value and size
647 
648  Parameters:
649  -----------
650  A: sparse matrix (d x K)
651  spatial components
652 
653  C: matrix or np.ndarray (K x T)
654  temporal components
655 
656  Returns:
657  -------
658  A_or: np.ndarray
659  ordered spatial components
660 
661  C_or: np.ndarray
662  ordered temporal components
663 
664  srt: np.ndarray
665  sorting mapping
666 
667  """
668  A = np.array(A.todense())
669  nA2 = np.sqrt(np.sum(A**2, axis=0))
670  K = len(nA2)
671  A = np.array(np.matrix(A) * spdiags(old_div(1, nA2), 0, K, K))
672  nA4 = np.sum(A**4, axis=0)**0.25
