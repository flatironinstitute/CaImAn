74 def extract_patch_coordinates(dims, rf, stride):
75  """
76  Function that partition the FOV in patches and return the indexed in 2D and 1D (flatten, order='F') formats
77  Parameters
78  ----------
79  dims: tuple of int
80  dimensions of the original matrix that will be divided in patches
81  rf: tuple of int
82  radius of receptive field, corresponds to half the size of the square patch
83  stride: tuple of int
84  degree of overlap of the patches
85  """
86  coords_flat = []
87  shapes = []
88  iters = [list(range(rf[i], dims[i] - rf[i], 2 * rf[i] - stride[i])) + [dims[i] - rf[i]]
89  for i in range(len(dims))]
90  coords = np.empty(list(map(len, iters)) + [len(dims)], dtype=np.object)
91  for count_0, xx in enumerate(iters[0]):
92  coords_x = np.arange(xx - rf[0], xx + rf[0] + 1)
93  coords_x = coords_x[(coords_x >= 0) & (coords_x < dims[0])]
94  for count_1, yy in enumerate(iters[1]):
95  coords_y = np.arange(yy - rf[1], yy + rf[1] + 1)
96  coords_y = coords_y[(coords_y >= 0) & (coords_y < dims[1])]
97  if len(dims) == 2:
98  idxs = np.meshgrid(coords_x, coords_y)
99  coords[count_0, count_1] = idxs
100  shapes.append(idxs[0].shape[::-1])
101 
102 
103  coords_ = np.ravel_multi_index(idxs, dims, order='F')
104  coords_flat.append(coords_.flatten())
105  else: # 3D data
106  for count_2, zz in enumerate(iters[2]):
107  coords_z = np.arange(zz - rf[2], zz + rf[2] + 1)
108  coords_z = coords_z[(coords_z >= 0) & (coords_z < dims[2])]
109  idxs = np.meshgrid(coords_x, coords_y, coords_z)
110  shps = idxs[0].shape
111  shapes.append([shps[1],shps[0],shps[2]])
112  coords[count_0, count_1, count_2] = idxs
113  coords_ = np.ravel_multi_index(idxs, dims, order='F')
114  coords_flat.append(coords_.flatten())
115 
116  for i, c in enumerate(coords_flat):
117  assert(len(c) == np.prod(shapes[i]))
118 
119  return map(np.sort, coords_flat), shapes
