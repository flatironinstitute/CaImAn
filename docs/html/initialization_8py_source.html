621 def hals(Y, A, C, b, f, bSiz=3, maxIter=5):
622  """ Hierarchical alternating least square method for solving NMF problem
623  Y = A*C + b*f
624 
625  input:
626  Y: d1 X d2 [X d3] X T, raw data. It will be reshaped to (d1*d2[*d3]) X T in this
627  function
628  A: (d1*d2[*d3]) X K, initial value of spatial components
629  C: K X T, initial value of temporal components
630  b: (d1*d2[*d3]) X nb, initial value of background spatial component
631  f: nb X T, initial value of background temporal component
632  bSiz: int or tuple of int
633  blur size. A box kernel (bSiz X bSiz [X bSiz]) (if int) or bSiz (if tuple) will
634  be convolved with each neuron's initial spatial component, then all nonzero
635  pixels will be picked as pixels to be updated, and the rest will be
636  forced to be 0.
637  maxIter: maximum iteration of iterating HALS.
638 
639  output:
640  the updated A, C, b, f
641 
642  Author: Johannes Friedrich, Andrea Giovannucci
643  """
644 
645  #%% smooth the components
646  dims, T = np.shape(Y)[:-1], np.shape(Y)[-1]
647  K = A.shape[1] # number of neurons
648  nb = b.shape[1] # number of background components
649  if isinstance(bSiz, (int, float)):
650  bSiz = [bSiz] * len(dims)
651  ind_A = nd.filters.uniform_filter(np.reshape(A, dims + (K,), order='F'), size=bSiz + [0])
652  ind_A = np.reshape(ind_A > 1e-10, (np.prod(dims), K), order='F')
653  ind_A = spr.csc_matrix(ind_A) # indicator of nonnero pixels
654 
655  def HALS4activity(Yr, A, C, iters=2):
656  U = A.T.dot(Yr)
657  V = A.T.dot(A)
658  for _ in range(iters):
659  for m in range(len(U)): # neurons and background
660  C[m] = np.clip(C[m] + (U[m] - V[m].dot(C)) / V[m, m], 0, np.inf)
661  return C
662 
663  def HALS4shape(Yr, A, C, iters=2):
664  U = C.dot(Yr.T)
665  V = C.dot(C.T)
666  for _ in range(iters):
667  for m in range(K): # neurons
668  ind_pixels = np.squeeze(ind_A[:, m].toarray())
669  A[ind_pixels, m] = np.clip(A[ind_pixels, m] +
670  ((U[m, ind_pixels] - V[m].dot(A[ind_pixels].T)) /
671  V[m, m]), 0, np.inf)
672  for m in range(nb): # background
673  A[:, K + m] = np.clip(A[:, K + m] + ((U[K + m] - V[K + m].dot(A.T)) /
674  V[K + m, K + m]), 0, np.inf)
675  return A
676 
677  Ab = np.c_[A, b]
678  Cf = np.r_[C, f.reshape(nb, -1)]
679  for _ in range(maxIter):
680  Cf = HALS4activity(np.reshape(Y, (np.prod(dims), T), order='F'), Ab, Cf)
681  Ab = HALS4shape(np.reshape(Y, (np.prod(dims), T), order='F'), Ab, Cf)
682 
683  return Ab[:, :-nb], Cf[:-nb], Ab[:, -nb:], Cf[-nb:].reshape(nb, -1)
